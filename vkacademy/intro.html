-- ========================================== -- 1. THE PURGE (Clean Slate) -- ========================================== DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users; DROP FUNCTION IF EXISTS public.handle_new_citizen(); DROP TABLE IF EXISTS public.master_roster CASCADE; DROP SEQUENCE IF EXISTS public.citizenship_seq; -- ========================================== -- 2. THE INFRASTRUCTURE -- ========================================== -- Create the sequencer for the cn- IDs CREATE SEQUENCE public.citizenship_seq START 1; -- Create the master roster table with strict rules CREATE TABLE public.master_roster ( id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE, citizenship_number TEXT UNIQUE NOT NULL, role TEXT NOT NULL DEFAULT 'Citizen', email TEXT UNIQUE NOT NULL, legal_first_name TEXT NOT NULL, legal_last_name TEXT NOT NULL, preferred_name TEXT NOT NULL, legal_birth_year INT NOT NULL, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW() ); -- ========================================== -- 3. ROW LEVEL SECURITY (RLS) -- ========================================== -- Turn on the security vault ALTER TABLE public.master_roster ENABLE ROW LEVEL SECURITY; -- Policy 1: Citizens can only read their own data CREATE POLICY "Citizens can view own record" ON public.master_roster FOR SELECT TO authenticated USING (auth.uid() = id); -- Policy 2: Citizens can update their own data (e.g., preferred name) CREATE POLICY "Citizens can update own record" ON public.master_roster FOR UPDATE TO authenticated USING (auth.uid() = id); -- Policy 3: Admins can see everything CREATE POLICY "Admins have full access" ON public.master_roster FOR ALL TO authenticated USING ( auth.uid() IN (SELECT id FROM public.master_roster WHERE role = 'Admin') ); -- ========================================== -- 4. THE BIRTH TRIGGER (Automation) -- ========================================== -- This function runs the millisecond a new user signs up CREATE OR REPLACE FUNCTION public.handle_new_citizen() RETURNS trigger AS $$ DECLARE new_cn TEXT; BEGIN -- Generate the next cn- ID new_cn := 'cn-' || nextval('public.citizenship_seq')::TEXT; -- Insert the exact data sent from the apply.html form INSERT INTO public.master_roster ( id, citizenship_number, role, email, legal_first_name, legal_last_name, preferred_name, legal_birth_year ) VALUES ( new.id, new_cn, 'Citizen', -- Default role new.email, new.raw_user_meta_data->>'first_name', new.raw_user_meta_data->>'last_name', new.raw_user_meta_data->>'preferred_name', (new.raw_user_meta_data->>'birth_year')::INT ); RETURN new; EXCEPTION WHEN OTHERS THEN -- If ANYTHING fails, it logs the exact error so we can read it, -- but allows the Auth user to be created so the app doesn't crash with a 500. RAISE LOG 'VANGUARD TRIGGER ERROR: % - %', SQLSTATE, SQLERRM; RETURN new; END; $$ LANGUAGE plpgsql SECURITY DEFINER; -- Attach the trigger to Supabase Auth CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE FUNCTION public.handle_new_citizen();
